pragma solidity ^0.4.18;

contract HashInput {
    
   string hash;
   uint CertNo;
   uint RevNo;
   string RevCertHash;
   uint Issuedto;
   uint IssuedBy;
   uint RevDate;
   
   
   function setCertHash(uint Cert_No,string _hash,uint Issued_to,uint Issued_By) public {
         CertNo=Cert_No;
         hash = _hash;
         Issuedto=Issued_to;
         IssuedBy=Issued_By;
     
   }
   
   function getCertHash() public constant returns (uint,string,uint,uint) {
       return (CertNo,hash,Issuedto,IssuedBy);
   }
   
    function setRevcert(uint Rev_No,string Rev_CertHash,uint Rev_Date) public {
       RevNo=Rev_No;
       RevCertHash=Rev_CertHash;
        RevDate=Rev_Date;
     
   }
   
   function getRevcert() public constant returns (uint,string,uint) {
       return (RevNo,RevCertHash,RevDate);
   }
    
}
